001package streak.schedulers;
002
003import javax.annotation.Nonnull;
004
005/**
006 * The scheduler is responsible for scheduling and executing tasks asynchronously.
007 * The scheduler provides an "abstract asynchronous boundary" to stream operators.
008 *
009 * <p>The scheduler has an internal clock that represents time as a monotonically increasing
010 * <code>int</code> value. The value may or may not have a direct relationship to wall-clock
011 * time and the unit of the value is defined by the implementation..</p>
012 */
013public interface Scheduler
014{
015  /**
016   * Return a value representing the "current time" of the scheduler.
017   *
018   * @return the "current time" of the scheduler.
019   */
020  int now();
021
022  /**
023   * Schedules the execution of the given task as soon as possible but still asynchronously.
024   *
025   * @param task the task to execute.
026   * @return the {@link SchedulerTask} instance that can be used to cancel execution of task.
027   */
028  @Nonnull
029  default SchedulerTask schedule( @Nonnull final Runnable task )
030  {
031    return schedule( task, 0 );
032  }
033
034  /**
035   * Schedules the execution of the given task after a specified delay.
036   *
037   * @param task  the task to execute.
038   * @param delay the delay before the task should execute.
039   * @return the {@link SchedulerTask} instance that can be used to cancel execution of task.
040   */
041  @Nonnull
042  default SchedulerTask schedule( @Nonnull final Runnable task, final int delay )
043  {
044    return schedule( task, delay, 0 );
045  }
046
047  /**
048   * Schedules the periodic execution of the given task with specified period.
049   *
050   * @param task   the task to execute.
051   * @param period the period after execution when the task should be re-executed. A negative value is invalid while a value of 0 indicates that the task is never rescheduled.
052   * @return the {@link SchedulerTask} instance that can be used to cancel execution of task.
053   */
054  @Nonnull
055  default SchedulerTask scheduleAtFixedRate( @Nonnull final Runnable task,
056                                             final int period )
057  {
058    return schedule( task, 0, period );
059  }
060
061  /**
062   * Schedules the periodic execution of the given task with specified period, after a specified delay.
063   *
064   * @param task         the task to execute.
065   * @param initialDelay the initial delay before the task should execute.
066   * @param period       the period after execution when the task should be re-executed. A negative value is invalid while a value of 0 indicates that the task is never rescheduled.
067   * @return the {@link SchedulerTask} instance that can be used to cancel execution of task.
068   */
069  @Nonnull
070  SchedulerTask schedule( @Nonnull Runnable task, int initialDelay, int period );
071
072  /**
073   * Initiate an orderly shutdown of the scheduler.
074   * No new tasks will be accepted but previously submitted tasks may be completed
075   * depending on the policy of the underlying scheduler.
076   *
077   * <p>Note: It is expected that this will be eliminated from the public interface exposed to user code
078   * and will only be visible internally to the framework.</p>
079   */
080  void shutdown();
081}
