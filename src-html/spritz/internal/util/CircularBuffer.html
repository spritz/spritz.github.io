001package spritz.internal.util;
002
003import java.util.Objects;
004import java.util.stream.Stream;
005import javax.annotation.Nonnull;
006import javax.annotation.Nullable;
007
008/**
009 * A circular buffer implementation.
010 */
011@SuppressWarnings( "Duplicates" )
012public final class CircularBuffer<T>
013{
014  /**
015   * The underlying object array.
016   */
017  private T[] _elements;
018  /**
019   * The pointer to first element.
020   */
021  private int _head;
022  /**
023   * The pointer to last element.
024   */
025  private int _tail;
026  /**
027   * Flag indicating whether buffer is wrapped around.
028   */
029  private boolean _isWrappedBuffer;
030
031  /**
032   * Create a buffer with specified initial capacity.
033   *
034   * @param initialCapacity the initial capacity of the buffer.
035   */
036  @SuppressWarnings( "unchecked" )
037  public CircularBuffer( final int initialCapacity )
038  {
039    assert initialCapacity > 0;
040    _elements = (T[]) new Object[ initialCapacity ];
041  }
042
043  public void clear()
044  {
045    _head = 0;
046    _tail = 0;
047    _isWrappedBuffer = false;
048  }
049
050  /**
051   * Return the current capacity of the buffer.
052   * The buffer may grow.
053   *
054   * @return the current capacity of the buffer.
055   */
056  public int getCapacity()
057  {
058    return _elements.length;
059  }
060
061  public int size()
062  {
063    if ( _isWrappedBuffer )
064    {
065      return _elements.length - _head + _tail;
066    }
067    else
068    {
069      return _tail - _head;
070    }
071  }
072
073  public void add( @Nonnull final T object )
074  {
075    tryGrowBeforeAdd();
076    insertLast( Objects.requireNonNull( object ) );
077  }
078
079  public void addFirst( @Nonnull final T object )
080  {
081    tryGrowBeforeAdd();
082    insertFirst( Objects.requireNonNull( object ) );
083  }
084
085  private void tryGrowBeforeAdd()
086  {
087    final int newSize = size() + 1;
088    if ( newSize > _elements.length )
089    {
090      grow( newSize );
091    }
092  }
093
094  @Nullable
095  public T get( final int index )
096  {
097    if ( index >= size() )
098    {
099      return null;
100    }
101    else
102    {
103      final int offset = ( _head + index ) % _elements.length;
104      return _elements[ offset ];
105    }
106  }
107
108  @Nullable
109  public T pop()
110  {
111    if ( isEmpty() )
112    {
113      return null;
114    }
115    final T result = _elements[ _head ];
116    _elements[ _head ] = null;
117
118    _head++;
119    if ( _head >= _elements.length )
120    {
121      _head = 0;
122      _isWrappedBuffer = false;
123    }
124
125    return result;
126  }
127
128  public boolean isEmpty()
129  {
130    return 0 == size();
131  }
132
133  public void truncate( final int size )
134  {
135    if ( _elements.length > size )
136    {
137      shrink( size );
138    }
139  }
140
141  /**
142   * Insert specified object at head of the buffer.
143   *
144   * @param object the object.
145   */
146  private void insertFirst( @Nonnull final T object )
147  {
148    if ( 0 == _head )
149    {
150      _head = _elements.length - 1;
151      _isWrappedBuffer = true;
152    }
153    else
154    {
155      _head--;
156    }
157    _elements[ _head ] = object;
158  }
159
160  /**
161   * Insert specified object at tail of the buffer.
162   *
163   * @param object the object.
164   */
165  private void insertLast( @Nonnull final T object )
166  {
167    _elements[ _tail ] = object;
168    _tail++;
169    if ( _tail >= _elements.length )
170    {
171      _tail = 0;
172      _isWrappedBuffer = true;
173    }
174  }
175
176  /**
177   * Increase the size of the buffer.
178   */
179  private void grow( final int minCapacity )
180  {
181    int newSize = _elements.length;
182    while ( newSize < minCapacity )
183    {
184      newSize = ( ( newSize - 1 ) * 2 ) + 1;
185    }
186
187    @SuppressWarnings( "unchecked" )
188    final T[] elements = (T[]) new Object[ newSize ];
189
190    int j = 0;
191    final int size = size();
192    for ( int i = 0; i < size; i++ )
193    {
194      final int index = ( _head + i ) % _elements.length;
195      elements[ j ] = _elements[ index ];
196      _elements[ index ] = null;
197      j++;
198    }
199
200    _elements = elements;
201    _head = 0;
202    _tail = j;
203    _isWrappedBuffer = false;
204  }
205
206  /**
207   * Decrease the size of the buffer.
208   */
209  private void shrink( final int maxCapacity )
210  {
211    @SuppressWarnings( "unchecked" )
212    final T[] elements = (T[]) new Object[ maxCapacity ];
213
214    int j = 0;
215    final int size = Math.min( size(), maxCapacity );
216    for ( int i = 0; i < size; i++ )
217    {
218      final int index = ( _head + i ) % _elements.length;
219      elements[ j ] = _elements[ index ];
220      _elements[ index ] = null;
221      j++;
222    }
223
224    _elements = elements;
225    _head = 0;
226    _tail = j;
227    _isWrappedBuffer = false;
228  }
229
230  /*
231   * This method is very inefficient and should only be used in invariant checking code.
232   */
233  public boolean contains( final T value )
234  {
235    final int size = size();
236    for ( int i = 0; i < size; i++ )
237    {
238      if ( value == get( i ) )
239      {
240        return true;
241      }
242    }
243    return false;
244  }
245
246  /*
247   * This method is very inefficient and should only be used in invariant checking code.
248   */
249  public Stream<T> stream()
250  {
251    final int size = size();
252    @SuppressWarnings( "unchecked" )
253    final T[] elements = (T[]) new Object[ size ];
254    for ( int i = 0; i < size; i++ )
255    {
256      elements[ i ] = get( i );
257    }
258    return Stream.of( elements );
259  }
260}
