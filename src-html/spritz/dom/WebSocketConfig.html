001package spritz.dom;
002
003import akasha.BinaryType;
004import java.util.Objects;
005import javax.annotation.Nonnull;
006import javax.annotation.Nullable;
007
008public final class WebSocketConfig
009{
010  @Nonnull
011  private final String _url;
012  @Nullable
013  private final String[] _protocols;
014  @Nullable
015  @BinaryType
016  private final String _binaryType;
017
018  public WebSocketConfig( @Nonnull final String url )
019  {
020    this( url, null, null );
021  }
022
023  public WebSocketConfig( @Nonnull final String url,
024                          @Nullable final String[] protocols,
025                          @BinaryType @Nullable final String binaryType )
026  {
027    _url = Objects.requireNonNull( url );
028    _protocols = protocols;
029    _binaryType = binaryType;
030  }
031
032  @Nonnull
033  public String getUrl()
034  {
035    return _url;
036  }
037
038  @Nullable
039  public String[] getProtocols()
040  {
041    return _protocols;
042  }
043
044  @Nullable
045  @BinaryType
046  public String getBinaryType()
047  {
048    return _binaryType;
049  }
050}
