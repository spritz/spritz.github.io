001package streak;
002
003import java.util.Collection;
004import java.util.concurrent.Callable;
005import java.util.function.Supplier;
006import java.util.stream.Collectors;
007import javax.annotation.Nonnull;
008import javax.annotation.Nullable;
009import streak.internal.annotations.DocCategory;
010import streak.internal.annotations.MetaDataSource;
011
012@MetaDataSource
013public final class Streak
014{
015  private Streak()
016  {
017  }
018
019  public static boolean shouldValidateSubscriptions()
020  {
021    return StreakConfig.shouldValidateSubscriptions();
022  }
023
024  /**
025   * Creates a stream that emits the parameters as items and then emits the completion signal.
026   *
027   * @param <T>    the type of items contained in the stream.
028   * @param values the values to emit.
029   * @return the new stream.
030   */
031  @SafeVarargs
032  @DocCategory( DocCategory.Type.CONSTRUCTION )
033  public static <T> Stream<T> of( @Nonnull final T... values )
034  {
035    return new StaticStreamSource<>( values );
036  }
037
038  /**
039   * Creates a stream that emits no items and immediately emits a completion signal.
040   *
041   * @param <T> the type of items that the stream declared as containing (despite never containing any items).
042   * @return the new stream.
043   */
044  @DocCategory( DocCategory.Type.CONSTRUCTION )
045  public static <T> Stream<T> empty()
046  {
047    return of();
048  }
049
050  /**
051   * Creates a stream that emits the value parameter as an item if the value is not null.
052   *
053   * @param <T>   the type of items contained in the stream.
054   * @param value the value to emit if non null.
055   * @return the new stream.
056   */
057  @DocCategory( DocCategory.Type.CONSTRUCTION )
058  public static <T> Stream<T> ofNullable( @Nullable final T value )
059  {
060    return null == value ? empty() : of( value );
061  }
062
063  /**
064   * Creates a stream that emits no items and immediately emits an error signal.
065   * This is an alias for {@link #error(Throwable)}.
066   *
067   * @param <T>   the type of items that the stream declared as containing (despite never containing any items).
068   * @param error the error to emit.
069   * @return the new stream.
070   * @see #error(Throwable)
071   */
072  @DocCategory( DocCategory.Type.CONSTRUCTION )
073  public static <T> Stream<T> fail( @Nonnull final Throwable error )
074  {
075    return new FailStreamSource<>( error );
076  }
077
078  /**
079   * Creates a stream that emits no items and immediately emits an error signal.
080   * This is an alias for {@link #fail(Throwable)}.
081   *
082   * @param <T>   the type of items that the stream declared as containing (despite never containing any items).
083   * @param error the error to emit.
084   * @return the new stream.
085   * @see #fail(Throwable)
086   */
087  @DocCategory( DocCategory.Type.CONSTRUCTION )
088  public static <T> Stream<T> error( @Nonnull final Throwable error )
089  {
090    return new FailStreamSource<>( error );
091  }
092
093  /**
094   * Creates a stream that emits items of the supplied collection.
095   *
096   * @param <T>    the type of items contained in the stream.
097   * @param values the collection of values to emit.
098   * @return the new stream.
099   */
100  @DocCategory( DocCategory.Type.CONSTRUCTION )
101  public static <T> Stream<T> fromCollection( @Nonnull final Collection<T> values )
102  {
103    return new CollectionStreamSource<>( values );
104  }
105
106  /**
107   * Creates a stream that emits items from the supplied {@link java.util.stream.Stream}.
108   *
109   * @param <T>    the type of items contained in the stream.
110   * @param stream the java.util.stream.Stream stream of values to emit.
111   * @return the new stream.
112   */
113  @DocCategory( DocCategory.Type.CONSTRUCTION )
114  public static <T> Stream<T> fromStream( @Nonnull final java.util.stream.Stream<T> stream )
115  {
116    return fromCollection( stream.collect( Collectors.toList() ) );
117  }
118
119  /**
120   * Creates an infinite stream that emits items from the {@link Callable} parameter.
121   * The user must be very careful to add a subsequent stream stage that cancels the stream
122   * otherwise this source will result in an infinite loop.
123   *
124   * @param <T>      the type of items contained in the stream.
125   * @param callable the function that generates values to emit.
126   * @return the new stream.
127   */
128  @DocCategory( DocCategory.Type.CONSTRUCTION )
129  public static <T> Stream<T> fromCallable( @Nonnull final Callable<T> callable )
130  {
131    return new GenerateStreamSource<>( callable );
132  }
133
134  /**
135   * Creates an infinite stream that emits items from the {@link Supplier} parameter.
136   * The user must be very careful to add a subsequent stream stage that cancels the stream
137   * otherwise this source will result in an infinite loop.
138   *
139   * @param <T>      the type of items contained in the stream.
140   * @param supplier the function that generates values to emit.
141   * @return the new stream.
142   */
143  @DocCategory( DocCategory.Type.CONSTRUCTION )
144  public static <T> Stream<T> fromSupplier( @Nonnull final Supplier<T> supplier )
145  {
146    return new GenerateStreamSource<>( supplier::get );
147  }
148
149  /**
150   * Creates a stream that completes when the {@link Runnable} parameter completes running.
151   * The stream will signal an error if the runnable generates an error while running.
152   *
153   * @param <T>      the type of items that the stream declared as containing (despite never containing any items).
154   * @param runnable the runnable to execute.
155   * @return the new stream.
156   */
157  @DocCategory( DocCategory.Type.CONSTRUCTION )
158  public static <T> Stream<T> fromRunnable( @Nonnull final Runnable runnable )
159  {
160    return new RunnableStreamSource<>( runnable );
161  }
162
163  /**
164   * Creates an infinite stream that emits items from the {@link Supplier} parameter at specified period.
165   * The user must be very careful to add a subsequent stream stage that cancels the stream
166   * otherwise this source will result in an infinite loop.
167   *
168   * @param <T>      the type of items contained in the stream.
169   * @param supplier the function that generates values to emit.
170   * @param period   the period with which items are emitted.
171   * @return the new stream.
172   */
173  @DocCategory( DocCategory.Type.CONSTRUCTION )
174  public static <T> Stream<T> generate( @Nonnull final Supplier<T> supplier, final int period )
175  {
176    return periodic( period ).map( e -> supplier.get() );
177  }
178
179  /**
180   * Creates a stream that emits no items, never completes and never fails.
181   *
182   * @param <T> the type of items that the stream declared as containing (despite never containing any items).
183   * @return the new stream.
184   */
185  @DocCategory( DocCategory.Type.CONSTRUCTION )
186  public static <T> Stream<T> never()
187  {
188    return new NeverStreamSource<>();
189  }
190
191  /**
192   * Create a stream that emits a sequence of numbers within a specified range.
193   * The stream create a sequence of [start, start + count).
194   *
195   * @param start the starting value of the range
196   * @param count the number of items to emit
197   * @return the new stream.
198   */
199  @DocCategory( DocCategory.Type.CONSTRUCTION )
200  public static Stream<Integer> range( final int start, final int count )
201  {
202    return new RangeStreamSource( start, count );
203  }
204
205  /**
206   * Create a stream that emits sequential numbers every specified interval of time.
207   * The stream create a sequence of [start, start + count).
208   *
209   * @param period the period with which items are emitted.
210   * @return the new stream.
211   */
212  @DocCategory( DocCategory.Type.CONSTRUCTION )
213  public static Stream<Integer> periodic( final int period )
214  {
215    return new PeriodicStreamSource( period );
216  }
217
218  @DocCategory( DocCategory.Type.CONSTRUCTION )
219  @SafeVarargs
220  public static <T> Stream<T> concat( @Nonnull final Stream<T>... upstreams )
221  {
222    return of( upstreams ).concatMap( v -> v );
223  }
224
225  @DocCategory( DocCategory.Type.CONSTRUCTION )
226  @SafeVarargs
227  public static <T> Stream<T> merge( @Nonnull final Stream<T>... upstreams )
228  {
229    return of( upstreams ).mergeMap( v -> v );
230  }
231}
