001package spritz.dom;
002
003import java.util.Objects;
004import javax.annotation.Nonnull;
005import javax.annotation.Nullable;
006
007public final class WebSocketConfig
008{
009  @Nonnull
010  private final String _url;
011  @Nullable
012  private final String[] _protocols;
013  @Nullable
014  private final String _binaryType;
015
016  public WebSocketConfig( @Nonnull final String url )
017  {
018    this( url, null, null );
019  }
020
021  public WebSocketConfig( @Nonnull final String url,
022                          @Nullable final String[] protocols,
023                          @Nullable final String binaryType )
024  {
025    _url = Objects.requireNonNull( url );
026    _protocols = protocols;
027    _binaryType = binaryType;
028  }
029
030  @Nonnull
031  public String getUrl()
032  {
033    return _url;
034  }
035
036  @Nullable
037  public String[] getProtocols()
038  {
039    return _protocols;
040  }
041
042  @Nullable
043  public String getBinaryType()
044  {
045    return _binaryType;
046  }
047}
