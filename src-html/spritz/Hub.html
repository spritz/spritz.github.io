001package spritz;
002
003import java.util.ArrayList;
004import java.util.HashSet;
005import java.util.Set;
006import javax.annotation.Nonnull;
007import javax.annotation.Nullable;
008import static org.realityforge.braincheck.Guards.*;
009
010public abstract class Hub<MessageInT, MessageOutT>
011  extends Stream<MessageOutT>
012  implements EventEmitter<MessageInT>
013{
014  private final Set<ForwardToEventEmitterSubscriber<MessageInT>> _upstreamSubscribers = new HashSet<>();
015  private final Set<DownstreamSubscription> _downstreamSubscriptions = new HashSet<>();
016  @Nullable
017  private Throwable _error;
018  private boolean _complete;
019
020  Hub( @Nullable final String name )
021  {
022    super( Spritz.areNamesEnabled() ? Stream.generateName( name, "subject" ) : null );
023  }
024
025  final void doSubscribe( @Nonnull Subscriber<? super MessageOutT> subscriber )
026  {
027    final DownstreamSubscription subscription = new DownstreamSubscription( subscriber );
028    subscriber.onSubscribe( subscription );
029    if ( subscription.isNotDone() )
030    {
031      completeSubscribe( subscription );
032    }
033    if ( subscription.isNotDone() )
034    {
035      if ( _complete )
036      {
037        subscriber.onComplete();
038      }
039      else if ( null != _error )
040      {
041        subscriber.onError( _error );
042      }
043      else
044      {
045        _downstreamSubscriptions.add( subscription );
046      }
047    }
048  }
049
050  void completeSubscribe( @Nonnull final DownstreamSubscription subscription )
051  {
052  }
053
054  /**
055   * {@inheritDoc}
056   */
057  @Override
058  public final void next( @Nonnull final MessageInT item )
059  {
060    if ( Spritz.shouldCheckApiInvariants() )
061    {
062      apiInvariant( () -> null == _error,
063                    () -> "Hub-0023: Hub.next(...) invoked after Hub.error(...) invoked." );
064      apiInvariant( () -> !_complete,
065                    () -> "Hub-0024: Hub.next(...) invoked after Hub.complete() invoked." );
066    }
067    performNext( item );
068  }
069
070  abstract void performNext( @Nonnull MessageInT item );
071
072  /**
073   * {@inheritDoc}
074   */
075  @Override
076  public final void error( @Nonnull final Throwable error )
077  {
078    if ( Spritz.shouldCheckApiInvariants() )
079    {
080      apiInvariant( () -> null == _error,
081                    () -> "Spritz-0025: Hub.error(...) invoked after Hub.error(...) invoked." );
082      apiInvariant( () -> !_complete,
083                    () -> "Spritz-0026: Hub.error(...) invoked after Hub.complete() invoked." );
084    }
085    _error = error;
086    performError( error );
087  }
088
089  abstract void performError( @Nonnull Throwable error );
090
091  /**
092   * {@inheritDoc}
093   */
094  @Override
095  public final void complete()
096  {
097    if ( Spritz.shouldCheckApiInvariants() )
098    {
099      invariant( () -> null == _error,
100                 () -> "Spritz-0027: Hub.complete(...) invoked after Hub.error(...) invoked." );
101      invariant( () -> !_complete,
102                 () -> "Spritz-0028: Hub.complete(...) invoked after Hub.complete() invoked." );
103    }
104    _complete = true;
105    performComplete();
106  }
107
108  abstract void performComplete();
109
110  void terminateUpstreamSubscribers()
111  {
112    for ( final ForwardToEventEmitterSubscriber<MessageInT> subscriber : new ArrayList<>( _upstreamSubscribers ) )
113    {
114      subscriber.cancel();
115    }
116    _upstreamSubscribers.clear();
117  }
118
119  @Nonnull
120  final ForwardToEventEmitterSubscriber<MessageInT> newUpstreamSubscriber()
121  {
122    final ForwardToEventEmitterSubscriber<MessageInT> subscriber = new ForwardToEventEmitterSubscriber<>( this );
123    _upstreamSubscribers.add( subscriber );
124    return subscriber;
125  }
126
127  @Nullable
128  final Throwable getError()
129  {
130    return _error;
131  }
132
133  /**
134   * {@inheritDoc}
135   */
136  @Override
137  public final boolean isDone()
138  {
139    return _complete || null != _error;
140  }
141
142  final boolean hasDownstreamSubscribers()
143  {
144    return !_downstreamSubscriptions.isEmpty();
145  }
146
147  void downstreamNext( @Nonnull final MessageOutT item )
148  {
149    for ( final DownstreamSubscription subscription : _downstreamSubscriptions )
150    {
151      subscription.getSubscriber().onNext( item );
152    }
153  }
154
155  void downstreamError( @Nonnull final Throwable error )
156  {
157    for ( final DownstreamSubscription subscription : _downstreamSubscriptions )
158    {
159      subscription.getSubscriber().onError( error );
160    }
161    _downstreamSubscriptions.clear();
162  }
163
164  void downstreamComplete()
165  {
166    for ( final DownstreamSubscription subscription : _downstreamSubscriptions )
167    {
168      subscription.getSubscriber().onComplete();
169    }
170    _downstreamSubscriptions.clear();
171  }
172
173  final class DownstreamSubscription
174    extends AbstractSubscription<MessageOutT>
175  {
176    DownstreamSubscription( @Nonnull final Subscriber<? super MessageOutT> subscriber )
177    {
178      super( subscriber );
179    }
180
181    @Override
182    void doCancel()
183    {
184      _downstreamSubscriptions.remove( this );
185    }
186
187    @Override
188    String getQualifiedName()
189    {
190      return Hub.this.getQualifiedName();
191    }
192  }
193}
