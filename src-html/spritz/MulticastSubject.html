001package spritz;
002
003import java.util.HashSet;
004import java.util.Objects;
005import java.util.Set;
006import javax.annotation.Nonnull;
007import javax.annotation.Nullable;
008import static org.realityforge.braincheck.Guards.*;
009
010/**
011 * A PublishOperator that emits items to currently subscribed {@link Subscriber} instances and terminal
012 * events to current or late {@link Subscriber} instances.
013 */
014public final class MulticastSubject<T>
015  extends AbstractEventEmitter<T>
016  implements Subject<T>
017{
018  /**
019   * The upstream stream stage.
020   */
021  @Nonnull
022  private final Publisher<T> _upstream;
023  @Nullable
024  private Subscription _upstreamSubscription;
025  private final Set<Subscriber<? super T>> _downstreamSubscribers = new HashSet<>();
026
027  /**
028   * Create a stream with specified upstream.
029   *
030   * @param upstream the upstream stream.
031   */
032  public MulticastSubject( @Nonnull final Publisher<T> upstream )
033  {
034    _upstream = Objects.requireNonNull( upstream );
035  }
036
037  /**
038   * Return the upstream stream.
039   *
040   * @return the upstream stream.
041   */
042  @Nonnull
043  protected final Publisher<T> getUpstream()
044  {
045    return _upstream;
046  }
047
048  /**
049   * {@inheritDoc}
050   */
051  @Override
052  public synchronized void subscribe( @Nonnull final Subscriber<? super T> subscriber )
053  {
054    if ( Spritz.shouldCheckApiInvariants() )
055    {
056      apiInvariant( () -> !_downstreamSubscribers.contains( subscriber ),
057                    () -> "Spritz-0023: Invoked MulticastSubject.subscribe(...) when subscriber is " +
058                          "already subscribed." );
059    }
060    if ( isComplete() )
061    {
062      subscriber.onComplete();
063    }
064    else if ( null != getError() )
065    {
066      subscriber.onError( getError() );
067    }
068    else
069    {
070      _downstreamSubscribers.add( subscriber );
071      final Subscription subscription = () -> {
072        _downstreamSubscribers.remove( subscriber );
073        if ( _downstreamSubscribers.isEmpty() )
074        {
075          disconnect();
076        }
077      };
078      subscriber.onSubscribe( subscription );
079      if ( 1 == _downstreamSubscribers.size() )
080      {
081        connect();
082      }
083    }
084  }
085
086  private synchronized void connect()
087  {
088    _upstream.subscribe( new Subscriber<T>()
089    {
090      @Override
091      public void onSubscribe( @Nonnull final Subscription subscription )
092      {
093        _upstreamSubscription = subscription;
094      }
095
096      @Override
097      public void onNext( @Nonnull final T item )
098      {
099        doNext( item );
100      }
101
102      @Override
103      public void onError( @Nonnull final Throwable error )
104      {
105        doError( error );
106      }
107
108      @Override
109      public void onComplete()
110      {
111        doComplete();
112      }
113    } );
114  }
115
116  private synchronized void disconnect()
117  {
118    assert null != _upstreamSubscription;
119    _upstreamSubscription.cancel();
120    _upstreamSubscription = null;
121  }
122
123  /**
124   * {@inheritDoc}
125   */
126  @Override
127  protected synchronized void doNext( @Nonnull final T item )
128  {
129    for ( final Subscriber<? super T> subscriber : _downstreamSubscribers )
130    {
131      subscriber.onNext( item );
132    }
133  }
134
135  /**
136   * {@inheritDoc}
137   */
138  @Override
139  protected synchronized void doError( @Nonnull final Throwable error )
140  {
141    for ( final Subscriber<? super T> subscriber : _downstreamSubscribers )
142    {
143      subscriber.onError( error );
144    }
145    _downstreamSubscribers.clear();
146  }
147
148  /**
149   * {@inheritDoc}
150   */
151  @Override
152  protected synchronized void doComplete()
153  {
154    for ( final Subscriber<? super T> subscriber : _downstreamSubscribers )
155    {
156      subscriber.onComplete();
157    }
158    _downstreamSubscribers.clear();
159  }
160}
